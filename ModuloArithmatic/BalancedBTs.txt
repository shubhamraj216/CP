long long findAns(long long n,long long i,long long prev,long long prevprev)
{ 
    long long ll=pow(10,9)+7;
    long long ans=((prev*prev)%ll+(2*prev*prevprev)%ll)%ll;
    if(i==n)
    {
        return ans;
    }
    prevprev=prev;
    prev=ans;
    return findAns(n,i+1,prev,prevprev);
}
